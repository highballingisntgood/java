text_area.addEventFilter(MouseEvent.MOUSE_PRESSED, event -> {
            if (event.getButton().equals(MouseButton.SECONDARY)) {
                Clipboard clipboard = Clipboard.getSystemClipboard();
                if (clipboard.hasString()) {
                    text_area.setOnContextMenuRequested(Event::consume);
                    text_area.appendText(clipboard.getString() + "\n");
                }
            }
        });

        text_area.addEventFilter(MouseEvent.MOUSE_RELEASED, event -> {
            String selectedText = text_area.getSelectedText();

            if (!selectedText.isEmpty()) {
                Clipboard clipboard = Clipboard.getSystemClipboard();
                ClipboardContent content = new ClipboardContent();
                content.putString(selectedText);
                clipboard.setContent(content);
            }
        });
